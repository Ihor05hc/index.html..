function removeElement(array, item) { // Знаходимо індекс елемента item у масиві const index = array.indexOf(item); // Якщо елемент знайдено, видаляємо його if (index !== -1) { array.splice(index, 1); // Видаляємо 1 елемент за індексом } } // Приклад використання const array = [1, 3, 4, 6, 2, 5, 7]; removeElement(array, 4); console.log(array); // Результат: [1, 3, 6, 2, 5, 7]
